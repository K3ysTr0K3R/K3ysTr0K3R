⚙ Optional GitHub Action: CVE Feed Update

".github/workflows/update-cves.yml"

name: Update CVE Feed

on:
  schedule:
    - cron: "0 */6 * * *"
  workflow_dispatch:

permissions:
  contents: write

jobs:
  update:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Update README CVE block
        run: |
          python3 - <<'PY'
          from pathlib import Path
          import re

          readme = Path("README.md")
          text = readme.read_text(encoding="utf-8")

          cves = [
              "CVE-2024-10914",
              "CVE-2024-27198",
              "CVE-2024-25600",
              "CVE-2024-4577",
              "CVE-2024-3273",
          ]

          block = "\n".join(f"- {cve}" for cve in cves)

          text = re.sub(
              r"<!-- START_CVE_FEED -->.*?<!-- END_CVE_FEED -->",
              f"<!-- START_CVE_FEED -->\n{block}\n<!-- END_CVE_FEED -->",
              text,
              flags=re.S,
          )

          readme.write_text(text, encoding="utf-8")
          PY

      - name: Commit changes
        run: |
          git config user.name "github-actions"
          git config user.email "github-actions@github.com"
          git add README.md
          git commit -m "Update CVE feed" || exit 0
          git push
